#include <stdio.h>
#include <time.h>
#include <stdlib.h>

void random(int number1, int number2) // fonksiyon tanýmý ve kullanýlacak deðerlerin cinsi
{
    int randomNo, counter;
    srand(time(NULL));    
    
    for(counter=0; counter<100; counter++) // 100 defa rasgele sayý seçiliyor
    {
        if(counter%10 == 0) { 
        printf("\n");   }  // 10. sayýdan sonra alt satýra geçiyor
        randomNo = rand()%(number2-number1) +number1; // rastgele seçilen sayý number1 ve number 2 arasýndaki bir deðere dönüþüyor
        printf("%d\t", randomNo);     // bu deðeri yazdýrýyor             
    }
    printf("\n");  // fonksiyondan sonra alt satýra geçiyor
   
}

int main() // main fonksiyonuna giriyor
{
    int a, b; // iki integer tanýmlýyor
    
    while(1){
        printf("enter N1:");
        scanf("%d", &a);         //verilen ilk integerý N1 e eþitliyor
        printf("enter N2:");     
        scanf("%d", &b);         //verilen ikinci integerý N2 ye eþitliyor
        
        if (a == 0 || b == 0) // N1 veya N2 "0" a eþitse döngüden çýkýyor
        {
           printf("\nN1 and N2 can't be zero, loop terminated");    
           break;
        }  
        else if(a < 0 || a > b )  // N1 negatifse yada N2 den büyükse yeni deðerler istiyor
        {
           printf("\nN1 can't be negative or bigger than N2 enter new values \n"); 
           continue;
        }
            
        else  // bu koþullarda deðilse random fonksiyonuna giriyor
           random(a, b); 
    }
    
    scanf("%d", &a);
    return 0;    
}
